☁️ Integração Google Drive / Sheets

Cole abaixo a URL do seu Google Apps Script Web App para salvar automaticamente os dados dos formulários no Google Sheets.
📁 Abrir pasta do Google Drive

Como configurar:
1. Acesse script.google.com e crie um novo projeto
2. Cole o código do Apps Script (disponível na pasta do Drive)
3. Clique em "Implantar" → "Novo implante" → Web App → "Qualquer pessoa"
4. Copie a URL gerada e cole abaixo
📋 Código para Google Apps Script

Copie este código, cole no Google Apps Script e implante como Web App.

const FOLDER_ID = '1Dv9mQUy7MlAeD2OGjT8dMm6yn3XgBJCL';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheetName = data.sheet || 'Dados';
    const row = data.row || [];
    
    // Get or create spreadsheet in folder
    const folder = DriveApp.getFolderById(FOLDER_ID);
    let ss;
    const files = folder.getFilesByName('Portal_T&S_Dados');
    if (files.hasNext()) {
      ss = SpreadsheetApp.open(files.next());
    } else {
      ss = SpreadsheetApp.create('Portal_T&S_Dados');
      DriveApp.getFileById(ss.getId()).moveTo(folder);
    }
    
    // Get or create sheet tab
    let sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
    }
    
    // Append row with timestamp
    sheet.appendRow([data.timestamp || new Date().toISOString(), ...row]);
    
    return ContentService
      .createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error',message:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
