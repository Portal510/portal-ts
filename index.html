☁️ Sincronização em Nuvem — Google Sheets

Configure uma vez e os formulários de qualquer dispositivo (celular, outro computador, etc.) atualizam o dashboard automaticamente.
✅ Solução gratuita — usa sua conta Google

Como configurar (5 minutos):
1. Acesse script.google.com e clique em "Novo projeto"
2. Apague o código de exemplo e cole o código abaixo (botão "Copiar")
3. Clique em Implantar → Novo implante
4. Tipo: App da Web → Executar como: Eu → Acesso: Qualquer pessoa
5. Clique em Implantar → autorize as permissões
6. Copie a URL da Web App gerada (começa com script.google.com/macros/s/...)
7. Cole a URL no campo abaixo e clique em 💾 Salvar

Após configurar: Todos os formulários enviam dados para o Google Sheets. O dashboard atualiza automaticamente a cada 15s buscando os dados do Sheets.
⬇️ Próximo passo — embutir URL no portal

Para funcionar em qualquer dispositivo, baixe o portal com a URL já embutida e suba no GitHub.

Copie este código, cole no Google Apps Script e implante como Web App.

// T&S Engenharia — Google Apps Script
// Cole em script.google.com → Implantar → App da Web
// Executar como: Eu  |  Acesso: Qualquer pessoa

const SS_NAME = 'Portal_TS_Dados';

function getOrCreateSS() {
  var files = DriveApp.getFilesByName(SS_NAME);
  if (files.hasNext()) return SpreadsheetApp.open(files.next());
  return SpreadsheetApp.create(SS_NAME);
}

function getOrCreateSheet(ss, name) {
  var sh = ss.getSheetByName(name);
  if (!sh) sh = ss.insertSheet(name);
  return sh;
}

// POST — recebe envios dos colaboradores
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = getOrCreateSS();
    var sh = getOrCreateSheet(ss, data.sheet || 'Dados');
    var entry = data.entry || {};
    sh.appendRow([data.ts || new Date().toISOString(), JSON.stringify(entry)]);
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// GET — retorna dados para o gestor (suporta JSONP para funcionar sem CORS)
function doGet(e) {
  var params = (e && e.parameter) ? e.parameter : {};
  var action = params.action || '';
  var callback = params.callback || '';
  var result;

  if (action !== 'pull') {
    result = {status:'ok', message:'T&S Engenharia API ativa'};
  } else {
    try {
      var ss = getOrCreateSS();
      result = {status:'ok', ts: new Date().toISOString()};
      var sheets = ['Ferias','Pausas','PendPausas','BH','Atestados',
                    'Dispositivos','Uniformes','Maleta','Feedback'];
      sheets.forEach(function(name) {
        var sh = ss.getSheetByName(name);
        if (!sh) { result[name] = []; return; }
        var rows = sh.getDataRange().getValues();
        var entries = [];
        rows.forEach(function(row) {
          if (!row[1]) return;
          try { var entry = JSON.parse(String(row[1])); if (entry && entry.id) entries.push(entry); } catch(ex) {}
        });
        result[name] = entries;
      });
    } catch(err) {
      result = {status:'error', message:err.message};
    }
  }

  var json = JSON.stringify(result);
  // JSONP — contorna CORS sem precisar de configuração extra
  if (callback) {
    return ContentService.createTextOutput(callback + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}
